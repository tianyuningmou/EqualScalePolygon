# EqualScalePolygon

#### Usage:
    from EqualScalePolygon import equal_scale_polygon
    
    print(equal_scale_polygon('0, 0; 0, 1; 1, 1; 1, 0', 1))
    print(equal_scale_polygon(((0, 0), (0, 1), (1, 1), (1, 0)), 1))
    print(equal_scale_polygon([(0, 0), (0, 1), (1, 1), (1, 0)], 1))
